059    package org.apache.webapp.balancer;
060    
061    import javax.servlet.http.HttpServletRequest;
062    
063    
064    /**
065     * This rule redirects requests if they
066     * are for a specific character encoding.
067     *
068     * @author Yoav Shapira
069     */
070    public class CharacterEncodingRule extends BaseRule {
071        /**
072         * The character encoding.
073         */
074        private String encoding;
075    
076        /**
077         * Sets the character encoding.
078         *
079         * @param theEncoding The encoding value
080         */
081        public void setEncoding(String theEncoding) {
082            if (theEncoding == null) {
083                throw new IllegalArgumentException("The encoding cannot be null.");
084            } else {
085                encoding = theEncoding;
086            }
087        }
088    
089        /**
090         * Returns the desired encoding.
091         *
092         * @return String
093         */
094        protected String getEncoding() {
095            return encoding;
096        }
097    
098        /**
099         * @see Rule#matches(HttpServletRequest request)
100         */
101        public boolean matches(HttpServletRequest request) {
102            String actualEncoding = request.getCharacterEncoding();
103    
104            return (getEncoding().compareTo(actualEncoding) == 0);
105        }
106    
107        /**
108         * Returns a String representation of this object.
109         *
110         * @return String
111         */
112        public String toString() {
113            StringBuffer buffer = new StringBuffer();
114    
115            buffer.append("[");
116            buffer.append(getClass().getName());
117            buffer.append(": ");
118    
119            buffer.append("Target encoding: ");
120            buffer.append(getEncoding());
121            buffer.append(" / ");
122    
123            buffer.append("Redirect URL: ");
124            buffer.append(getRedirectUrl());
125    
126            buffer.append("]");
127    
128            return buffer.toString();
129        }
130    }
