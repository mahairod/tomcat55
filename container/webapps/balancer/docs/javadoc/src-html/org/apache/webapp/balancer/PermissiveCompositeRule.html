001    package org.apache.webapp.balancer;
002    
003    import java.util.Iterator;
004    
005    import javax.servlet.http.HttpServletRequest;
006    
007    /**
008     * A permissive composite rule implements
009     * a logical OR operation on its rules:
010     * any one rule can match for overall success.
011     *
012     * @author Yoav Shapira
013     */
014    public class PermissiveCompositeRule extends CompositeRule {
015        /**
016         * Constructor.
017         */
018        public PermissiveCompositeRule() {
019            super();
020        }
021    
022        /**
023         * @see Rule#matches(HttpServletRequest)
024         */
025        public boolean matches(HttpServletRequest request) {
026            Iterator iter = getRuleIterator();
027    
028            Rule currentRule = null;
029            boolean currentMatches = true;
030            
031            while(iter.hasNext()) {
032                currentRule = (Rule) iter.next();
033                currentMatches = currentRule.matches(request);
034                
035                if(currentMatches) {
036                    return true;
037                }
038            }
039    
040            return false;
041        }
042    }
