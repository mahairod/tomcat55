059    package org.apache.webapp.balancer;
060    
061    import javax.servlet.http.HttpServletRequest;
062    
063    
064    /**
065     * This rule checks for the presence
066     * of a specific request header, optionally
067     * with a specific value.  The value may
068     * be null.
069     *
070     * @author Yoav Shapira
071     */
072    public class RequestHeaderRule extends BaseRule {
073        /**
074         * The header name, cannot be null.
075         */
076        private String headerName;
077    
078        /**
079         * The header value.  This may be
080         * null to indicate any value is OK.
081         */
082        private String headerValue;
083    
084        /**
085         * Sets the header name.
086         *
087         * @param theName The name
088         */
089        public void setHeaderName(String theName) {
090            if (theName == null) {
091                throw new IllegalArgumentException(
092                    "The header name cannot be null.");
093            } else {
094                headerName = theName;
095            }
096        }
097    
098        /**
099         * Returns the header name to match.
100         *
101         * @return The header name
102         */
103        protected String getHeaderName() {
104            return headerName;
105        }
106    
107        /**
108         * Sets the header value.
109         *
110         * @param theValue The header value
111         */
112        public void setHeaderValue(String theValue) {
113            headerValue = theValue;
114        }
115    
116        /**
117         * Returns the desired header value,
118         * which may be null.
119         *
120         * @return String
121         */
122        protected String getHeaderValue() {
123            return headerValue;
124        }
125    
126        /**
127         * @see Rule#matches(HttpServletRequest request)
128         */
129        public boolean matches(HttpServletRequest request) {
130            String actualHeaderValue = request.getHeader(getHeaderName());
131    
132            if (actualHeaderValue == null) {
133                return (getHeaderValue() == null);
134            } else {
135                return (actualHeaderValue.compareTo(getHeaderValue()) == 0);
136            }
137        }
138    
139        /**
140         * Returns a String representation of this object.
141         *
142         * @return String
143         */
144        public String toString() {
145            StringBuffer buffer = new StringBuffer();
146    
147            buffer.append("[");
148            buffer.append(getClass().getName());
149            buffer.append(": ");
150    
151            buffer.append("Header name: ");
152            buffer.append(getHeaderName());
153            buffer.append(" / ");
154    
155            buffer.append("Header value: ");
156            buffer.append(getHeaderValue());
157            buffer.append(" / ");
158    
159            buffer.append("Redirect URL: ");
160            buffer.append(getRedirectUrl());
161    
162            buffer.append("]");
163    
164            return buffer.toString();
165        }
166    }
